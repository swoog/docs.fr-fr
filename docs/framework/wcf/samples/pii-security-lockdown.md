## <a name="locking-configuration-file-elements"></a><span data-ttu-id="04fbf-136">Verrouillage des éléments de fichier de configuration</span><span class="sxs-lookup"><span data-stu-id="04fbf-136">Locking configuration file elements</span></span>

               <serviceDebug includeExceptionDetailInFaults="False" />   
             </behavior>   
          </serviceBehaviors>   
       </behaviors>   
    </system.serviceModel>
